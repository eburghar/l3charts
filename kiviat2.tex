\documentclass{article}
\usepackage{tikz}
\usepackage{expl3,xparse}
\usepackage{fontspec}
\usepackage[default]{raleway}

\usetikzlibrary{shapes}

% \renewcommand*\familydefault{\ttdefault}% Force the monospace version of any font used

\ExplSyntaxOn
\NewDocumentEnvironment {kiviatchart} {m O{3.5cm} O{3.5cm} O{5}} {
	\clist_set:Nn \l_labels_clist {#1}
	\tl_set:Nx \labels {\clist_use:Nn \l_labels_clist {,}}
	\tl_set:Nx \D {\clist_count:N \l_labels_clist}% number of dimensions (config option)
	\tl_set:Nn \R {#2}% maximal diagram radius (config option)
	\tl_set:Nn \L {#3}% radius to put dimension labels (config option)
	\tl_set:Nn \U {#4}% number of scale units (config option)
	\tl_set:Nx \A {360/\D}% calculated angle between dimension axes

	\NewDocumentEnvironment {set} {} {
		\seq_new:N \l_values_seq
		\RenewDocumentCommand \value {m} {
			\seq_put_right:Nn \l_values_seq {####1}
		}
	}{
		\tl_set:Nx \values {\seq_use:Nn \l_values_seq {,}}
		\seq_get_left:NN \l_values_seq \firstval

		\foreach \v [count=\i from 1] in \values {
			\node at (D\i-\v) [fill,circle,draw,inner\c_space_tl sep=1pt] {};
		}

		\draw [color=black,line\c_space_tl width=1.5pt,opacity=1,fill\c_space_tl opacity=0.3,fill=gray] (D1-\firstval) \foreach \v [count=\i from 1] in \values {-- (D\i-\v)} -- cycle;
	}

	\begin{tikzpicture}[scale=1]
		\path (0\c_colon_str 0cm) coordinate (O);% define coordinate for origin

		% draw the spiderweb
		\foreach \X in {1,...,\D} {
				\draw[opacity=0.8] (\X*\A \c_colon_str 0) -- (\X*\A \c_colon_str \R);
			}

		\foreach \Y in {0,...,\U}{
				% dimensions axes
				\foreach \X in {1,...,\D}{
						\path (\X*\A \c_colon_str \Y*\R/\U) coordinate (D\X-\Y);
						% \fill (D\X-\Y) circle (1pt); % intersection points on the web
					}
				% units polygones
				\draw[opacity=0.2] (0 \c_colon_str \Y*\R/\U) \foreach \X in {1,...,\D}{
						-- (\X*\A \c_colon_str \Y*\R/\U)
					} -- cycle;
			}

		% units labels
		\foreach \Y in {1,...,\U}{
				\node[opacity=0.8,below] at (D1-\Y) {\tt\tiny\Y};
			}

		% define labels for each dimension axis (names config option)
		\foreach \l [count=\i from 1] in \labels {
			\path (\i*\A \c_colon_str \L) node {\colorbox{black}{\textcolor{white}{\l}}};
		}
} {
	\end{tikzpicture}
}
\ExplSyntaxOff
